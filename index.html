                    </button>


                    {isModalOpen && (

                        <div className="fixed inset-0 z-50 flex items-center justify-center p-4">

                                    <LucideIcon name="x" size={20} />

                                </button>

                                <h2 className="text-3xl font-black mb-2">{editingId ? '修改成就' : '記錄新成就'}</h2>

                                <p className="text-gray-400 mb-6 font-medium">理財是一種生活的藝術。</p>

                                <form onSubmit={handleSubmit} className="space-y-4">

                                        <div className="col-span-2">

                                            <label className="block text-xs font-black text-gray-400 uppercase mb-2 ml-1">投資平台</label>

                                        </div>

                                        <div>

                                                {CURRENCIES.map(c => <option key={c.code} value={c.code}>{c.code}</option>)}

                                            </select>

                                        </div>

                                        <div>

                                        </div>

                                    </div>

                                    <div>

                                        <label className="block text-xs font-black text-gray-400 uppercase mb-2 ml-1">投資類型</label>

                                        <input type="text" name="type" value={formData.type} onChange={handleInputChange} placeholder="例如：成長股..." className="w-full px-6 py-3.5 bg-gray-50 border-none rounded-3xl focus:ring-2 focus:ring-black font-bold" />

                                    </div>

                                    <div>

                                        <label className="block text-xs font-black text-gray-400 uppercase mb-2 ml-1">備註</label>

                                    </div>

                                    <div>

                                            {COLOR_OPTIONS.map((color) => (

                                                <button key={color.id} type="button" onClick={() => setFormData(p => ({...p, colorId: color.id}))}

                                                    className={`w-9 h-9 rounded-xl border-2 transition-all ${formData.colorId === color.id ? 'border-black scale-110 shadow-lg' : 'border-transparent opacity-60 hover:opacity-100'} ${color.class.split(' ')[0]}`} />

                                            ))}

                                        </div>

                                    </div>

                                    <div className="flex gap-4 pt-4">

                                        {editingId && (

                                            <button type="button" onClick={(e) => handleDelete(e, editingId)} className="flex-1 bg-red-50 text-red-500 py-4 rounded-3xl font-black text-lg hover:bg-red-100 transition-all flex items-center justify-center gap-2">

                                                <LucideIcon name="trash-2" size={20} /> 刪除

                                            </button>

                                        )}

                                        <button type="submit" className="flex-[2] bg-black text-white py-4 rounded-3xl font-black text-lg hover:shadow-xl active:scale-95 transition-all">

                                            {editingId ? '完成修改' : '確認新增'}

                                        </button>

                                    </div>

                                </form>

                            </div>

                        </div>

                    )}

                </div>

            );
